-- ───────────────────────────────────────────── -- VIEWS for dashboard queries -- ───────────────────────────────────────────── CREATE OR REPLACE VIEW v_user_stats AS SELECT u.id AS user_id, u.full_name, u.email, u.plan, COUNT(DISTINCT bd.id) AS active_bots, COALESCE(SUM(bd.total_profit), 0) AS total_profit, COALESCE(SUM(bd.trades_count), 0) AS total_trades, COALESCE(SUM(bd.win_count), 0) AS total_wins, CASE WHEN SUM(bd.trades_count) > 0 THEN ROUND(SUM(bd.win_count) / SUM(bd.trades_count) * 100, 2) ELSE 0 END AS win_rate_pct FROM users u LEFT JOIN bot_deployments bd ON bd.user_id = u.id AND bd.status = 'running' GROUP BY u.id; -- Monthly revenue view CREATE OR REPLACE VIEW v_monthly_revenue AS SELECT DATE_FORMAT(paid_at, '%Y-%m') AS month, method, COUNT(*) AS payment_count, SUM(amount) AS total_revenue FROM payments WHERE status = 'succeeded' GROUP BY DATE_FORMAT(paid_at, '%Y-%m'), method ORDER BY month DESC;
